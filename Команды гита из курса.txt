mkdir - новая директория
cd    - перемещение по пути
сd .. - вернуться на папку выше
echo "..." > name_file.extension  - добавить строку ... в файл name_file.extension


git init                     - иниц-ть папку с гит репозиторием
git add name_file            - добавить в staging area (индекс) файлы к коммиту

git commit -m "text_commit"  - добавить коммит с текстом "text_commit"
git log                      - посмотреть текущие коммиты

git checkout часть_sha_ключа - переместиться на коммит по части sha_ключа (первые 3-5 символов)
git checkout название_ветки  - переместиться в ветку 

git branch                   - увидеть все ветки репозитория
git branch название_ветки    - создать ветку название_ветки

git branch -m название_ветки1 - сменить название текущей ветки на название_ветки1
git branch -d название_ветки  - удалить ветку название_ветки (текущую ветку удалить нельзя)     

git merge -m "текст_коммита" название_ветки  - выполнить слияние текущей ветки с название_ветки с текстом коммита "текст_коммита"
